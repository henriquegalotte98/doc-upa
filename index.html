const API_URL =
  location.hostname === "localhost"
    ? "http://localhost:8080/api"
    : "https://SEU-BACKEND.up.railway.app/api";

async function apiRequest(path, options = {}) {
  const response = await fetch(`${API_URL}${path}`, {
    headers: { "Content-Type": "application/json" },
    ...options
  });

  if (!response.ok) throw new Error("Erro na operação.");
  return response.status === 204 ? null : response.json();
}

let ultimaNarrada = null;

async function atualizarPainel() {
  const chamada = await apiRequest("/chamadas/ultima");
  if (!chamada) return;

  if (chamada.id !== ultimaNarrada) {
    ultimaNarrada = chamada.id;

    const voz = new SpeechSynthesisUtterance(
      `Paciente ${chamada.atendimento.paciente.nome}, dirija-se a ${chamada.sala}.`
    );

    voz.lang = "pt-BR";
    speechSynthesis.speak(voz);
  }
}

setInterval(atualizarPainel, 2000);
