Integrante 5 — Back-End Usuários

Visão do produto

Problema

Organizar recepção, triagem, fila e atendimento.

Público

Recepcionista, enfermeiro, médico e administrador.

Objetivo

Entregar um MVP fullstack pequeno e funcional.

Apresentação

Simular o fluxo em computadores diferentes.

Escopo do MVP

Incluído

  • Login simples.
  • CRUD de funcionários.
  • Cadastro de atendimento.
  • Triagem.
  • Painel com narração.
  • Prontuário simplificado.

Não incluído

  • JWT.
  • Recuperação de senha.
  • Integração com SUS.
  • IA médica.
  • Relatórios complexos.

const API_URL =
  location.hostname === "localhost" || location.hostname === "127.0.0.1"
    ? "http://localhost:8080/api"
    : "https://SEU-BACKEND.up.railway.app/api";

const session = {
  getUser() {
    return JSON.parse(localStorage.getItem("usuario"));
  },
  setUser(user) {
    localStorage.setItem("usuario", JSON.stringify(user));
  },
  clear() {
    localStorage.removeItem("usuario");
  }
};

async function api(path, options = {}) {
  const response = await fetch(`${API_URL}${path}`, {
    headers: { "Content-Type": "application/json" },
    ...options
  });

  if (!response.ok) {
    throw new Error("Não foi possível concluir a operação.");
  }

  if (response.status === 204) return null;
  return response.json();
}

function redirecionarPorCargo(cargo) {
  const rotas = {
    ADMIN: "dashboard.html",
    RECEPCIONISTA: "balcao.html",
    ENFERMEIRO: "triagem.html",
    MEDICO: "consultorio.html"
  };

  location.href = rotas[cargo] || "login.html";
}

          method: "POST",

<!DOCTYPE html>
<html lang="pt-BR">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>Painel de Chamadas</title>
  <link rel="stylesheet" href="style.css">
</head>
<body>
  <main class="container">
    <section class="card">
      <p>Paciente chamado</p>
      <h1 id="senha">Aguardando</h1>
      <h2 id="nome">—</h2>
      <p id="destino">—</p>
    </section>
  </main>

  <script src="script.js"></script>
  <script>
    let ultimaNarrada = null;

    function narrar(chamada) {
      const texto =
        `Paciente ${chamada.atendimento.paciente.nome}, ` +
        `senha ${chamada.atendimento.senha}, ` +
        `dirija-se a ${chamada.sala}.`;

      speechSynthesis.cancel();

      const mensagem = new SpeechSynthesisUtterance(texto);
      mensagem.lang = "pt-BR";
      mensagem.rate = 0.9;

      speechSynthesis.speak(mensagem);
    }

    async function atualizarPainel() {
      const chamada = await api("/chamadas/ultima");
      if (!chamada) return;

      document.getElementById("senha").textContent = chamada.atendimento.senha;
      document.getElementById("nome").textContent =
        chamada.atendimento.paciente.nome;
      document.getElementById("destino").textContent = chamada.sala;

      if (chamada.id !== ultimaNarrada) {
        ultimaNarrada = chamada.id;
        narrar(chamada);
      }
    }

    atualizarPainel();
    setInterval(atualizarPainel, 2000);
  </script>
</body>
</html>

@RestController
@RequestMapping("/api/auth")
public class AuthController {

  private final UsuarioRepository repository;

  public AuthController(UsuarioRepository repository) {
    this.repository = repository;
  }

  @PostMapping("/login")
  public Usuario login(@RequestBody LoginRequest request) {
    return repository
      .findByEmailAndSenhaAndAtivoTrue(request.email(), request.senha())
      .orElseThrow(() -> new IllegalArgumentException("Login inválido."));
  }
}

package com.sistemaupa.controller;

import org.springframework.web.bind.annotation.*;
import java.util.List;

@RestController
@RequestMapping("/api/usuarios")
public class UsuarioController {

  private final UsuarioRepository repository;

  public UsuarioController(UsuarioRepository repository) {
    this.repository = repository;
  }

  @GetMapping
  public List<Usuario> listar() {
    return repository.findAll();
  }

  @PostMapping
  public Usuario criar(@RequestBody Usuario usuario) {
    usuario.setAtivo(true);
    return repository.save(usuario);
  }

  @PutMapping("/{id}")
  public Usuario atualizar(@PathVariable Long id, @RequestBody Usuario dados) {
    Usuario usuario = repository.findById(id)
      .orElseThrow(() -> new IllegalArgumentException("Usuário não encontrado."));

    usuario.setNome(dados.getNome());
    usuario.setEmail(dados.getEmail());
    usuario.setCargo(dados.getCargo());
    usuario.setAtivo(dados.isAtivo());

    return repository.save(usuario);
  }

  @DeleteMapping("/{id}")
  public void excluir(@PathVariable Long id) {
    repository.deleteById(id);
  }
}
